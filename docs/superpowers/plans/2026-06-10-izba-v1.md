```rust
use serde::{de::DeserializeOwned, Serialize};
use std::io::{Read, Write};

pub const MAX_FRAME: u32 = 16 * 1024 * 1024;

#[derive(Debug, thiserror::Error)]
pub enum FrameError {
    #[error("clean EOF before frame")]
    Eof,
    #[error("frame of {0} bytes exceeds limit")]
    TooLarge(u32),
    #[error("io: {0}")]
    Io(#[from] std::io::Error),
    #[error("json: {0}")]
    Json(#[from] serde_json::Error),
}

pub fn write_frame<W: Write, T: Serialize>(w: &mut W, msg: &T) -> Result<(), FrameError> {
    let payload = serde_json::to_vec(msg)?;
    let len = u32::try_from(payload.len()).map_err(|_| FrameError::TooLarge(u32::MAX))?;
    if len > MAX_FRAME { return Err(FrameError::TooLarge(len)); }
    w.write_all(&len.to_le_bytes())?;
    w.write_all(&payload)?;
    w.flush()?;
    Ok(())
}

pub fn read_frame<R: Read, T: DeserializeOwned>(r: &mut R) -> Result<T, FrameError> {
    let mut len_buf = [0u8; 4];
    // First byte distinguishes clean EOF from truncation.
    match r.read(&mut len_buf[..1])? {
        0 => return Err(FrameError::Eof),
        _ => r.read_exact(&mut len_buf[1..])?,
    }
    let len = u32::from_le_bytes(len_buf);
    if len > MAX_FRAME { return Err(FrameError::TooLarge(len)); }
    let mut payload = vec![0u8; len as usize];
    r.read_exact(&mut payload)?;
    Ok(serde_json::from_slice(&payload)?)
}
```
`lib.rs`: `pub mod codec; pub use codec::{read_frame, write_frame, FrameError, MAX_FRAME};`

### Task 5: izba-core — OCI layer flatten (tar-merge)

The most bug-prone pure component. Whiteout semantics per OCI image-spec: basename `.wh.<name>` deletes `<name>` from lower layers; `.wh..wh..opq` makes the directory opaque (hides all lower content). Output: one merged tar (entries lexicographically sorted so parents precede children; hardlink entries emitted after everything else), data streamed from per-layer temp tar files via `raw_file_position()` — host FS never sees unpacked files.

1. `merge_disjoint`: two layers, different files → both present.
2. `upper_overrides`: same path in both → upper's content and mode win.
3. `whiteout_deletes`: layer2 has `Whiteout("etc/passwd")` → absent in output, and the `.wh.` entry itself absent.

```rust
use std::collections::BTreeMap;
use std::io::{Read, Seek, SeekFrom, Write};

pub fn flatten_layers<W: Write>(layer_readers: Vec<Box<dyn Read>>, out: W) -> anyhow::Result<()> {
    // Pass 0: decompress each layer (gzip; pass through if not gzipped — sniff magic 0x1f 0x8b)
    //         to a tempfile::tempfile() as plain tar.
    // Pass 1: index. For each layer i (0 = lowest), iterate entries with
    //         Archive::new(&mut f).entries()?; record per entry:
    //           path (normalized, no trailing '/'), EntryKey { layer: i, offset: e.raw_file_position(),
    //           size: e.size(), header: e.header().clone() }.
    //         Apply onto a BTreeMap<String, Node> "tree" with overlay semantics:
    //           - ".wh..wh..opq" basename => remove all existing keys with prefix "dir/" (keep dir itself); mark nothing in output.
    //           - ".wh.<name>" basename  => remove "parent/name" and all keys under "parent/name/".
    //           - regular entry          => if an existing entry at that path is a dir and new one is not
    //                                       (or vice versa), first remove the subtree; then insert,
    //                                       remembering (layer, offset, header).
    //         Within one layer, later duplicate paths simply overwrite (tar last-wins).
    // Pass 2: emit. BTreeMap iteration is lexicographic => parents (e.g. "a") sort before
    //         children ("a/b") because '/' (0x2f) is low; verified by test 8.
    //         Split into non-hardlinks then hardlinks. For each: write the stored header
    //         (Builder::append with the original header preserves mode/uid/gid/mtime/linkname),
    //         streaming file data by seeking the layer tempfile to offset and io::copy size bytes.
    //         PAX/long-name entries: use Builder::append_data with the entry's path when the
    //         original header can't carry it; rely on tar crate's automatic GNU long-name handling.
}
```
Implementation detail that WILL bite: `entries()` must be consumed in order, and `raw_file_position()` is the position of the entry's data block in the (decompressed) tar — reopen the tempfile by `try_clone()` for pass 2 seeks. Normalize paths: strip leading `./` and `/`; skip the bare `.` entry; reject paths containing `..` (return error — hostile image).

```rust
pub fn spawn_detached(cmd: &CommandSpec, log: &Path) -> anyhow::Result<PidIdentity> {
    let logf = File::options().create(true).append(true).open(log)?;
    let mut c = Command::new(&cmd.argv[0]);
    c.args(&cmd.argv[1..])
     .stdin(Stdio::null()).stdout(logf.try_clone()?).stderr(logf);
    unsafe { c.pre_exec(|| { nix::unistd::setsid().map_err(io::Error::from)?; Ok(()) }); }
    let child = c.spawn()?;
    let pid = child.id();
    // do NOT keep `child` — std::process::Child would try to reap; intentionally forget it:
    std::mem::forget(child);
    Ok(PidIdentity { pid, starttime: proc_starttime(pid)? })
}
fn proc_starttime(pid: u32) -> anyhow::Result<u64> {
    let s = fs::read_to_string(format!("/proc/{pid}/stat"))?;
    let after = s.rfind(')').context("bad stat")? + 2;          // comm can contain spaces/parens
    let fields: Vec<&str> = s[after..].split(' ').collect();
    Ok(fields[19].parse()?)                                      // field 22 overall = index 19 after comm
}
pub fn pid_alive(id: &PidIdentity) -> bool { proc_starttime(id.pid).map(|s| s == id.starttime).unwrap_or(false) }
pub fn kill_pid(id: &PidIdentity) -> anyhow::Result<()> { /* SIGKILL via nix::sys::signal::kill if pid_alive */ }
```
